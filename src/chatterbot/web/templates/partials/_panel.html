Usage: {% from "partials/_panel.html" import panel %} {% call panel('starred', 'Your favorites are here', icon='fa-star', count=starred_active|length) %} …content… {% endcall %} The driver in _panels_driver.html attaches to any element marked `data-insights-panels` and applies the streamer's saved order + collapse state on first render and after every HTMX swap. #} {% macro panel(panel_id, title, icon='fa-circle', count=None, icon_tone='accent', subtitle=None) %}
